def rms
        total = 0
        if @rms.nil?
               @data.each do |point|
                       diff = (point[2] - point[3])
                       total += (diff * diff) 
               end
               @rms = Math.sqrt(total / @data.length)
        end
        return @rms
end

def stddev
        total = 0
        if @stddev.nil?
               @data.each do |point|
                       diff = ((point[2] - point[3]).abs - @mean)
                       total += (diff * diff) 
               end
               @stddev = Math.sqrt(total / @data.length)
        end
        return @stddev
end
